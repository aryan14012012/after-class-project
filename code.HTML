def option_one(): """Function to handle Option 1.""" print(">> You chose Option 1. Performing action one.") def option_two(): """Function to handle Option 2.""" print(">> You chose Option 2. Performing action two.") def main_menu(): """Displays the menu and handles user input.""" while True: # Loop indefinitely until the user chooses to exit print("\nChoose from the following options:") print("1) Option One") print("2) Option Two") print("3) Exit") choice = input("Enter your choice (1, 2, or 3): ") if choice == '1': option_one() elif choice == '2': option_two() elif choice == '3': print(">> Exiting the program. Goodbye!") break # Exit the while loop else: # Handle invalid input print(">> Invalid input! Please enter a number between 1 and 3.") if __name__ == "__main__": main_menu()
